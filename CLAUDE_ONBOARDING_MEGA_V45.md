  // Actually in AE: r proportional to (90-lat). Equator = lat 0, r = 90/90*maxR.
  // Let's recalculate: r = (90 - lat) / 90 * maxR. lat=0 → r=maxR. That puts equator at edge.
  // Standard AE: equator is halfway. Let North Pole = center, edge = equator at r = maxR*0.5,
  // then southern lats extend beyond.
  // Let me use: r = colatitude / 180 * maxR so that south pole is at edge.
  // colatitude of equator = 90 → r = 90/180 * maxR = maxR/2

  // Longitude lines
  for(let lon = 0; lon < 360; lon += 30){
    let a = (lon-90)*Math.PI/180;
    ctx.beginPath();
    ctx.moveTo(cx,cy);
    ctx.lineTo(cx + maxR*Math.cos(a), cy + maxR*Math.sin(a));

function raDecToDomePos(raHours, decDeg){
  // In dome model: dec maps to distance from center axis
  // RA maps to azimuthal angle
  // Stars are embedded IN the firmament at H = dome surface
  
  // Colatitude from north pole
  const colat = (90 - decDeg) * Math.PI / 180;
  const azimuth = (raHours / 24) * 2 * Math.PI;
  
  // Use dome surface: spheroid shape scaled by H_POLARIS/DOME_R in Y
  // On a hemisphere of radius DOME_R, the point at (colat, azimuth) is:
  const r = DOME_R;
  const x = r * Math.sin(colat) * Math.cos(azimuth);
  const z = r * Math.sin(colat) * Math.sin(azimuth);
  const y = r * Math.cos(colat) * (H_POLARIS / DOME_R); // scale Y to dome height
  
  return new THREE.Vector3(x, Math.max(y, 0), z);
}

function updateRays(sunPos){
  while(rayGroup.children.length) rayGroup.remove(rayGroup.children[0]);
  
  // Shoot 6 rays from Sun bending down via aetheric refraction
  for(let i = 0; i < 6; i++){
    const angle = (i / 6) * Math.PI * 2;
    const targetR = DOME_R * 0.4;
    const tx = targetR * Math.cos(angle);
    const tz = targetR * Math.sin(angle);
    
    const pts = [];
    for(let j = 0; j <= 80; j++){
      const t = j / 80;
      const x = sunPos.x + (tx - sunPos.x) * t;
      const z = sunPos.z + (tz - sunPos.z) * t;
      // Aetheric bending: power curve simulating exponential density gradient
      const y = H_SUN * Math.pow(1 - t, 2.8);
      pts.push(new THREE.Vector3(x, y, z));
    }
    const lineGeo = new THREE.BufferGeometry().setFromPoints(pts);
    const lineMat = new THREE.LineBasicMaterial({
      color: new THREE.Color().setHSL(0.08 + i * 0.015, 1, 0.55),
      transparent: true, opacity: 0.35,
      blending: THREE.AdditiveBlending
    });
    rayGroup.add(new THREE.Line(lineGeo, lineMat));
  }
}

    sunR * Math.cos(sunAngle),
    H_SUN, // FIXED altitude — never fluctuates
    sunR * Math.sin(sunAngle)
  );

  // ── Moon Position ──
  // 27.3 day synodic period
  const moonAngle = (simMinutes / (1440 * 27.3)) * 2 * Math.PI;
  const moonR = SUN_R_EQUINOX * 0.95;
  moonGroup.position.set(
    moonR * Math.cos(moonAngle),
    H_MOON, // FIXED altitude
    moonR * Math.sin(moonAngle)
  );

  // Update lensing rays
  if(rayGroup.visible && Math.floor(simMinutes) % 5 === 0){
    updateRays(sunGroup.position);
  }

  // ── HUD Updates ──
  const totalMinutes = Math.floor(simMinutes);
  const hrs = Math.floor((totalMinutes % 1440) / 60);
  const mins = totalMinutes % 60;
  const secs = Math.floor((simMinutes % 1) * 60);
  document.getElementById('clock').textContent =
    String(hrs).padStart(2,'0') + ':' + String(mins).padStart(2,'0') + ':' + String(secs).padStart(2,'0');
  document.getElementById('day-count').textContent =
    `Day ${Math.floor(dayFrac)} · Dome Rotation: ${(domeAngle * 180 / Math.PI % 360).toFixed(1)}°`;

  ctrl.update();
  renderer.render(scene, camera);
